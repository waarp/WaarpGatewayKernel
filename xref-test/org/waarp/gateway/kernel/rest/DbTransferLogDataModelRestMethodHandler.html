43  public class DbTransferLogDataModelRestMethodHandler extends DataModelRestMethodHandler<DbTransferLog> {
44      public static final String BASEURI = "logs";
45  
46      public static enum FILTER_ARGS {
47          MODETRANS("MODE TRANS name subtext"),
48          ACCOUNTID("ACCOUNT id information subtext"),
49          USERID("USER id information subtext"),
50          FILENAME("FILENAME information subtext"),
51          INFOSTATUS("Info status information subtext");
52  
53          public String type;
54  
55          FILTER_ARGS(String type) {
56              this.type = type;
57          }
58      }
59  
60      public String user;
61      public String account;
62  
63      /**
64       * @param name
65       * @param config
66       * @param method
67       */
68      public DbTransferLogDataModelRestMethodHandler(RestConfiguration config, METHOD... method) {
69          super(BASEURI, config, method);
70      }
71  
72      @Override
73      protected void checkAuthorization(HttpRestHandler handler, RestArgument arguments, RestArgument result,
74              METHOD method) throws HttpForbiddenRequestException {
75          // valid all
76      }
77  
78      @Override
79      protected DbTransferLog getItem(HttpRestHandler handler, RestArgument arguments, RestArgument result, Object body)
80              throws HttpIncorrectRequestException, HttpInvalidAuthenticationException, HttpNotFoundRequestException {
81          ObjectNode arg = arguments.getUriArgs().deepCopy();
82          arg.setAll(arguments.getBody());
83          try {
84              JsonNode node = RestArgument.getId(arg);
85              long id;
86              if (node.isMissingNode()) {
87                  // shall not be but continue however
88                  id = arg.path(DbTransferLog.Columns.SPECIALID.name()).asLong();
89              } else {
90                  id = node.asLong();
91              }
92              return new DbTransferLog(handler.getDbSession(), user, account, id);
93          } catch (WaarpDatabaseException e) {
94              throw new HttpNotFoundRequestException("Issue while reading from database " + arg, e);
95          }
96      }
97  
98      @Override
99      protected DbTransferLog createItem(HttpRestHandler handler, RestArgument arguments, RestArgument result,
100             Object body) throws HttpIncorrectRequestException, HttpInvalidAuthenticationException {
101         ObjectNode arg = arguments.getUriArgs().deepCopy();
102         arg.setAll(arguments.getBody());
103         try {
104             DbTransferLog newlog = new DbTransferLog(handler.getDbSession());
105             newlog.setFromJson(arg, false);
106             if (newlog.getAccount() == null || newlog.getUser() == null
107                     || newlog.getSpecialId() == DbConstant.ILLEGALVALUE) {
108                 throw new WaarpDatabaseSqlException("Not enough argument to create the object");
109             }
110             newlog.insert();
111             return newlog;
112         } catch (WaarpDatabaseException e) {
113             throw new HttpIncorrectRequestException("Issue while inserting into database", e);
114         }
115     }
116 
117     @Override
118     protected DbPreparedStatement getPreparedStatement(HttpRestHandler handler, RestArgument arguments,
119             RestArgument result, Object body) throws HttpIncorrectRequestException,
120             HttpInvalidAuthenticationException {
121         ObjectNode arg = arguments.getUriArgs().deepCopy();
122         arg.setAll(arguments.getBody());
123         String modetrans = arg.path(FILTER_ARGS.MODETRANS.name()).asText();
124         if (modetrans == null || modetrans.isEmpty()) {
125             modetrans = null;
126         }
127         String accountid = arg.path(FILTER_ARGS.ACCOUNTID.name()).asText();
128         if (accountid == null || accountid.isEmpty()) {
129             accountid = null;
130         }
131         String userid = arg.path(FILTER_ARGS.USERID.name()).asText();
132         if (userid == null || userid.isEmpty()) {
133             userid = null;
134         }
135         String filename = arg.path(FILTER_ARGS.FILENAME.name()).asText();
136         if (filename == null || filename.isEmpty()) {
137             filename = null;
138         }
139         String infostatus = arg.path(FILTER_ARGS.INFOSTATUS.name()).asText();
140         if (infostatus == null || infostatus.isEmpty()) {
141             infostatus = null;
142         }
143         try {
144             return DbTransferLog.getFilterPrepareStament(handler.getDbSession(),
145                     modetrans, accountid, userid, filename, infostatus);
146         } catch (WaarpDatabaseNoConnectionException e) {
147             throw new HttpIncorrectRequestException("Issue while reading from database", e);
148         } catch (WaarpDatabaseSqlException e) {
149             throw new HttpIncorrectRequestException("Issue while reading from database", e);
150         }
151     }
152 
153     @Override
154     protected DbTransferLog getItemPreparedStatement(DbPreparedStatement statement)
155             throws HttpIncorrectRequestException, HttpNotFoundRequestException {
156         try {
157             return DbTransferLog.getFromStatement(statement);
158         } catch (WaarpDatabaseNoConnectionException e) {
159             throw new HttpIncorrectRequestException("Issue while selecting from database", e);
160         } catch (WaarpDatabaseSqlException e) {
161             throw new HttpNotFoundRequestException("Issue while selecting from database", e);
162         }
163     }
164 
165     @Override
166     public String getPrimaryPropertyName() {
167         return Columns.SPECIALID.name();
168     }
169 
170     @Override
171     protected ArrayNode getDetailedAllow() {
172         ArrayNode node = JsonHandler.createArrayNode();
173 
174         ObjectNode node1 = JsonHandler.createObjectNode();
175         node1.put(DbTransferLog.JSON_MODEL, DbTransferLog.class.getSimpleName());
176         DbTransferLog empty = new DbTransferLog(null);
177         DbValue[] values = empty.getAllFields();
178         for (DbValue dbValue : values) {
179             node1.put(dbValue.getColumn(), dbValue.getType());
180         }
181 
182         ObjectNode node2, node3;
183         if (this.methods.contains(METHOD.GET)) {
184             node2 = RestArgument.fillDetailedAllow(
185                     METHOD.GET,
186                     this.path + "/id",
187                     COMMAND_TYPE.GET.name(),
188                     JsonHandler.createObjectNode().put(DbTransferLog.Columns.SPECIALID.name(),
189                             "SPECIALID as Long in URI as " + this.path + "/id"),
190                     node1);
191             node.add(node2);
192 
193             node3 = JsonHandler.createObjectNode();
194             for (FILTER_ARGS arg : FILTER_ARGS.values()) {
195                 node3.put(arg.name(), arg.type);
196             }
197             node2 = RestArgument.fillDetailedAllow(METHOD.GET, this.path, COMMAND_TYPE.MULTIGET.name(),
198                     node3, JsonHandler.createArrayNode().add(node1));
199             node.add(node2);
200         }
201         if (this.methods.contains(METHOD.PUT)) {
202             node3 = JsonHandler.createObjectNode();
203             node3.put(DbTransferLog.Columns.HOSTID.name(), "SPECIALID as Long in URI as " + this.path + "/id");
204             for (DbValue dbValue : values) {
205                 if (dbValue.getColumn().equalsIgnoreCase(DbTransferLog.Columns.SPECIALID.name())) {
206                     continue;
207                 }
208                 node3.put(dbValue.getColumn(), dbValue.getType());
209             }
210             node2 = RestArgument.fillDetailedAllow(METHOD.PUT, this.path + "/id", COMMAND_TYPE.UPDATE.name(),
211                     node3, node1);
212             node.add(node2);
213         }
214         if (this.methods.contains(METHOD.DELETE)) {
215             node3 = JsonHandler.createObjectNode();
216             node3.put(DbTransferLog.Columns.SPECIALID.name(), "SPECIALID as Long in URI as " + this.path + "/id");
217             node2 = RestArgument.fillDetailedAllow(METHOD.DELETE, this.path + "/id", COMMAND_TYPE.DELETE.name(),
218                     node3, node1);
219             node.add(node2);
220         }
221         if (this.methods.contains(METHOD.POST)) {
222             node3 = JsonHandler.createObjectNode();
223             for (DbValue dbValue : values) {
224                 node3.put(dbValue.getColumn(), dbValue.getType());
225             }
226             node2 = RestArgument.fillDetailedAllow(METHOD.POST, this.path, COMMAND_TYPE.CREATE.name(),
227                     node3, node1);
228             node.add(node2);
229         }
230         node2 = RestArgument.fillDetailedAllow(METHOD.OPTIONS, this.path, COMMAND_TYPE.OPTIONS.name(), null, null);
231         node.add(node2);
232 
233         return node;
234     }

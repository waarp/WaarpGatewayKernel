36  public abstract class HttpBusinessFactory {
37      public static final HttpDataFactory factory = new DefaultHttpDataFactory(
38              DefaultHttpDataFactory.MINSIZE); // Disk if size exceed MINSIZE =
39      // 16K
40      // Disk if size exceed MINSIZE = 16K, but for FileUpload from Ark directly
41      // XXX FIXME TODO to setup outside !
42      public static String TempPath = "J:/GG/ARK/TMP"; // "C:/Temp/Java/GG/ARK/TMP";
43  
44      /**
45       * Initialize the Disk support
46       */
47      public static void initialize(String tempPath) {
48          TempPath = tempPath;
49          DiskFileUpload.deleteOnExitTemporaryFile = true; // should delete file
50                                                           // on exit (in normal
51                                                           // exit)
52          DiskFileUpload.baseDirectory = TempPath; // system temp
53                                                   // directory
54          DiskAttribute.deleteOnExitTemporaryFile = true; // should delete file on
55                                                          // exit (in normal exit)
56          DiskAttribute.baseDirectory = TempPath; // system temp directory
57      }
58  
59      /**
60       * It returns the AbstractHttpBusinessRequest to use during a new request.
61       * 
62       * Note that fields given in parameter should be updated according to their values if needed.
63       * 
64       * @param remoteAddress
65       *            the remote SocketAddress in use
66       * @param fields
67       *            the fields linked hashmap (to preserver order) to set for the new request
68       * @param page
69       *            source HttpPage
70       * @return the AbstractHttpBusinessRequest to use during a new request
71       */
72      public abstract AbstractHttpBusinessRequest getNewHttpBusinessRequest(
73              SocketAddress remoteAddress,
74              LinkedHashMap<String, AbstractHttpField> fields, HttpPage page);
75  
76      /**
77       * 
78       * @param pages
79       * @param title
80       * @param clasz
81       * @return True if the default error pages are correctly added
82       */
83      public static boolean addDefaultErrorPages(HttpPageHandler pages, String title, Class<?> clasz) {
84          String pagename, header, footer, beginform, endform, nextinform, uri, errorpage, classname;
85          PageRole pageRole;
86          LinkedHashMap<String, AbstractHttpField> linkedHashMap;
87          String fieldname, fieldinfo, fieldvalue;
88          FieldRole fieldRole;
89          boolean fieldvisibility, fieldmandatory, fieldcookieset, fieldtovalidate;
90          int fieldrank;
91  
92          // Need as default error pages: 400, 401, 403, 404, 406, 500
93          try {
94              pageRole = PageRole.ERROR;
95              pagename = "400";
96              uri = "400";
97              header = "<HTML><HEAD><TITLE>" + title + "</TITLE></HEAD><BODY>";
98              footer = "</BODY></HTML>";
99              beginform = "<table border=\"0\"><tr><td><h1>" + title + "</h1>";
100             endform = "</td></tr></table><br><CENTER><HR WIDTH=\"75%\" NOSHADE color=\"blue\"></CENTER>";
101             nextinform = "</td></tr><tr><td>";
102             classname = clasz.getName();
103             errorpage = null;
104             linkedHashMap = new LinkedHashMap<String, AbstractHttpField>();
105             fieldname = AbstractHttpField.ERRORINFO;
106             fieldinfo = "Error information";
107             fieldvalue = HttpResponseStatus.BAD_REQUEST.reasonPhrase();
108             fieldRole = FieldRole.BUSINESS_INPUT_TEXT;
109             fieldvisibility = true;
110             fieldmandatory = false;
111             fieldcookieset = false;
112             fieldtovalidate = false;
113             fieldrank = 1;
114             linkedHashMap.put(fieldname, new DefaultHttpField(fieldname, fieldRole, fieldinfo,
115                     fieldvalue,
116                     fieldvisibility, fieldmandatory, fieldcookieset, fieldtovalidate,
117                     FieldPosition.BODY, fieldrank));
118             pages.getHashmap().put(uri, new HttpPage(pagename, null, header, footer, beginform, endform,
119                     nextinform,
120                     uri, pageRole, errorpage, classname, linkedHashMap));
121 
122             pagename = "401";
123             uri = "401";
124             linkedHashMap = new LinkedHashMap<String, AbstractHttpField>();
125             fieldvalue = HttpResponseStatus.UNAUTHORIZED.reasonPhrase();
126             linkedHashMap.put(fieldname, new DefaultHttpField(fieldname, fieldRole, fieldinfo,
127                     fieldvalue,
128                     fieldvisibility, fieldmandatory, fieldcookieset, fieldtovalidate,
129                     FieldPosition.BODY, fieldrank));
130             pages.getHashmap().put(uri, new HttpPage(pagename, null, header, footer, beginform, endform,
131                     nextinform,
132                     uri, pageRole, errorpage, classname, linkedHashMap));
133 
134             pagename = "403";
135             uri = "403";
136             linkedHashMap = new LinkedHashMap<String, AbstractHttpField>();
137             fieldvalue = HttpResponseStatus.FORBIDDEN.reasonPhrase();
138             linkedHashMap.put(fieldname, new DefaultHttpField(fieldname, fieldRole, fieldinfo,
139                     fieldvalue,
140                     fieldvisibility, fieldmandatory, fieldcookieset, fieldtovalidate,
141                     FieldPosition.BODY, fieldrank));
142             pages.getHashmap().put(uri, new HttpPage(pagename, null, header, footer, beginform, endform,
143                     nextinform,
144                     uri, pageRole, errorpage, classname, linkedHashMap));
145 
146             pagename = "404";
147             uri = "404";
148             linkedHashMap = new LinkedHashMap<String, AbstractHttpField>();
149             fieldvalue = HttpResponseStatus.NOT_FOUND.reasonPhrase();
150             linkedHashMap.put(fieldname, new DefaultHttpField(fieldname, fieldRole, fieldinfo,
151                     fieldvalue,
152                     fieldvisibility, fieldmandatory, fieldcookieset, fieldtovalidate,
153                     FieldPosition.BODY, fieldrank));
154             pages.getHashmap().put(uri, new HttpPage(pagename, null, header, footer, beginform, endform,
155                     nextinform,
156                     uri, pageRole, errorpage, classname, linkedHashMap));
157 
158             pagename = "406";
159             uri = "406";
160             linkedHashMap = new LinkedHashMap<String, AbstractHttpField>();
161             fieldvalue = HttpResponseStatus.NOT_ACCEPTABLE.reasonPhrase();
162             linkedHashMap.put(fieldname, new DefaultHttpField(fieldname, fieldRole, fieldinfo,
163                     fieldvalue,
164                     fieldvisibility, fieldmandatory, fieldcookieset, fieldtovalidate,
165                     FieldPosition.BODY, fieldrank));
166             pages.getHashmap().put(uri, new HttpPage(pagename, null, header, footer, beginform, endform,
167                     nextinform,
168                     uri, pageRole, errorpage, classname, linkedHashMap));
169 
170             pagename = "500";
171             uri = "500";
172             linkedHashMap = new LinkedHashMap<String, AbstractHttpField>();
173             fieldvalue = HttpResponseStatus.INTERNAL_SERVER_ERROR.reasonPhrase();
174             linkedHashMap.put(fieldname, new DefaultHttpField(fieldname, fieldRole, fieldinfo,
175                     fieldvalue,
176                     fieldvisibility, fieldmandatory, fieldcookieset, fieldtovalidate,
177                     FieldPosition.BODY, fieldrank));
178             pages.getHashmap().put(uri, new HttpPage(pagename, null, header, footer, beginform, endform,
179                     nextinform,
180                     uri, pageRole, errorpage, classname, linkedHashMap));
181             return true;
182         } catch (ClassNotFoundException e) {
183         } catch (InstantiationException e) {
184         } catch (IllegalAccessException e) {
185         }
186         return false;
187     }

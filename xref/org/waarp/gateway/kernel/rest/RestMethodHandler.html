48  /**
49   * Rest Method handler (used by Http Rest Handler)
50   * 
51   * @author "Frederic Bregier"
52   *
53   */
54  public abstract class RestMethodHandler {
55      /**
56       * Internal Logger
57       */
58      private static final WaarpLogger logger = WaarpLoggerFactory
59              .getLogger(RestMethodHandler.class);
60  
61      protected final String name;
62      protected final String path;
63      protected final Set<METHOD> methods;
64      protected final boolean isBodyJsonDecode;
65      protected final RestConfiguration restConfiguration;
66  
67      /**
68       * @param name
69       *            name associated with this Method Handler (to enable some HashMap or Enum classification)
70       * @param path
71       *            associated base Path
72       * @param isBodyJsonDecode
73       *            Is this method Handler using a Json as Body
74       * @param config
75       *            the associated configuration
76       * @param method
77       *            the associated methods
78       */
79      public RestMethodHandler(String name, String path, boolean isBodyJsonDecode, RestConfiguration config,
80              METHOD... method) {
81          this.name = name;
82          this.path = path;
83          this.methods = new HashSet<HttpRestHandler.METHOD>();
84          setMethods(method);
85          setMethods(METHOD.OPTIONS);
86          this.isBodyJsonDecode = isBodyJsonDecode;
87          this.restConfiguration = config;
88      }
89  
90      protected void setMethods(METHOD... method) {
91          for (METHOD method2 : method) {
92              methods.add(method2);
93          }
94      }
95  
96      /**
97       * Will assign the intersection of both set of Methods
98       * 
99       * @param selectedMethods
100      *            the selected Methods among available
101      * @param validMethod
102      *            the validMethod for this handler
103      */
104     protected void setIntersectionMethods(METHOD[] selectedMethods, METHOD... validMethod) {
105         Set<METHOD> set = new HashSet<METHOD>();
106         for (METHOD method : validMethod) {
107             set.add(method);
108         }
109         Set<METHOD> set2 = new HashSet<METHOD>();
110         for (METHOD method : selectedMethods) {
111             set2.add(method);
112         }
113         set.retainAll(set2);
114         METHOD[] methodsToSet = set.toArray(new METHOD[0]);
115         setMethods(methodsToSet);
116     }
117 
118     public String getName() {
119         return name;
120     }
121 
122     public String getPath() {
123         return path;
124     }

127      *

131     public boolean isMethodIncluded(METHOD method) {
132         return methods.contains(method);

158     /**
159      * Get data from BODY (supposedly a Json)

168     public abstract Object getBody(HttpRestHandler handler, ByteBuf body, RestArgument arguments, RestArgument result)
169             throws HttpIncorrectRequestException;
170 
171     /**
172      * Called when all Data were passed to the handler
173      * 
174      * @param handler
175      * @param arguments
176      * @param result
177      * @param body
178      * @throws HttpIncorrectRequestException
179      * @throws HttpNotFoundRequestException
180      */
181     public abstract void endParsingRequest(HttpRestHandler handler, RestArgument arguments, RestArgument result,
182             Object body) throws HttpIncorrectRequestException, HttpInvalidAuthenticationException,
183             HttpNotFoundRequestException;
184 
185     /**
186      * Called when an exception occurs
187      * 
188      * @param handler
189      * @param arguments
190      * @param result
191      * @param body
192      * @param exception
193      * @return the status to used in sendReponse
194      */
195     public HttpResponseStatus handleException(HttpRestHandler handler, RestArgument arguments, RestArgument result,
196             Object body, Exception exception) {
197         if (exception instanceof HttpInvalidAuthenticationException) {
198             result.setResult(HttpResponseStatus.UNAUTHORIZED);
199             return HttpResponseStatus.UNAUTHORIZED;
200         } else if (exception instanceof HttpForbiddenRequestException) {
201             result.setResult(HttpResponseStatus.FORBIDDEN);
202             return HttpResponseStatus.FORBIDDEN;
203         } else if (exception instanceof HttpIncorrectRequestException) {
204             result.setResult(HttpResponseStatus.BAD_REQUEST);
205             return HttpResponseStatus.BAD_REQUEST;
206         } else if (exception instanceof HttpMethodNotAllowedRequestException) {
207             result.setResult(HttpResponseStatus.METHOD_NOT_ALLOWED);
208             return HttpResponseStatus.METHOD_NOT_ALLOWED;
209         } else if (exception instanceof HttpNotFoundRequestException) {
210             result.setResult(HttpResponseStatus.NOT_FOUND);
211             return HttpResponseStatus.NOT_FOUND;
212         } else {
213             result.setResult(HttpResponseStatus.INTERNAL_SERVER_ERROR);
214             return HttpResponseStatus.INTERNAL_SERVER_ERROR;
215         }
216     }
217 
218     /**
219      * Send a response (correct or not)
220      * 
221      * @param handler
222      * @param ctx
223      * @param arguments
224      * @param result
225      * @param body
226      * @param status
227      * @return The ChannelFuture if this response will need the channel to be closed, else null
228      */
229     public abstract ChannelFuture sendResponse(HttpRestHandler handler, ChannelHandlerContext ctx,
230             RestArgument arguments, RestArgument result, Object body, HttpResponseStatus status);
231 
232     protected ChannelFuture sendOptionsResponse(HttpRestHandler handler, ChannelHandlerContext ctx,
233             RestArgument result, HttpResponseStatus status) {
234         String list = result.getAllowOption();
235         String answer = result.toString();
236         ByteBuf buffer = Unpooled.wrappedBuffer(answer.getBytes(WaarpStringUtils.UTF8));
237         HttpResponse response = handler.getResponse(buffer);
238         if (status == HttpResponseStatus.UNAUTHORIZED) {
239             ChannelFuture future = ctx.writeAndFlush(response);
240             return future;
241         }
242         response.headers().add(HttpHeaderNames.CONTENT_TYPE, "application/json");
243         response.headers().add(HttpHeaderNames.REFERER, handler.getRequest().uri());
244         response.headers().add(HttpHeaderNames.ALLOW, list);
245         logger.debug("Msg ready");
246         ChannelFuture future = ctx.writeAndFlush(response);
247         if (handler.isWillClose()) {
248             System.err.println("Will close session in RestMethodHandler");
249             return future;
250         }
251         return null;
252     }

254     /**
255      * Options command that all handler should implement
256      * 
257      * @param handler
258      * @param arguments
259      * @param result
260      */
261     protected void optionsCommand(HttpRestHandler handler, RestArgument arguments, RestArgument result) {
262         result.setCommand(COMMAND_TYPE.OPTIONS);
263         METHOD[] realmethods = METHOD.values();
264         boolean[] allMethods = new boolean[realmethods.length];
265         for (METHOD methoditem : methods) {
266             allMethods[methoditem.ordinal()] = true;
267         }
268         String allow = null;
269         for (int i = 0; i < allMethods.length; i++) {
270             if (allMethods[i]) {
271                 if (allow == null) {
272                     allow = realmethods[i].name();
273                 } else {
274                     allow += "," + realmethods[i].name();
275                 }
276             }
277         }
278         result.addOptions(allow, path, getDetailedAllow());
279     }

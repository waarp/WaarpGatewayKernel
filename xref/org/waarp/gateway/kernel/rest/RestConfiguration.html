21  package org.waarp.gateway.kernel.rest;
22  
23  import java.io.File;
24  import java.io.IOException;
25  import java.util.HashMap;
26  import java.util.Map.Entry;
27  
28  import org.waarp.common.crypto.HmacSha256;
29  import org.waarp.common.exception.CryptoException;
30  import org.waarp.common.utility.WaarpStringUtils;
31  
32  /**
33   * General RestConfiguration model
34   * @author "Frederic Bregier"
35   *
36   */
37  public class RestConfiguration {
38  	public static enum CRUD {
39  		CREATE(0x01),
40  		READ(0x02),
41  		UPDATE(0x04),
42  		DELETE(0x08),
43  		ALL(0x0F);
44  		
45  		public byte mask;
46  		CRUD(int mask) {
47  			this.mask = (byte) mask;
48  		}
49  		
50  		public boolean isValid(byte tocheck) {
51  			return (tocheck & mask) != 0;
52  		}
53  	}
54  	/**
55  	 * SERVER REST interface using explicit address (null means all available)
56  	 */
57  	public String REST_ADDRESS = null;
58  	/**
59  	 * Http REST port (SSL or not SSL)
60  	 */
61  	public int REST_PORT = -1;
62  	/**
63  	 * SERVER REST interface using SSL
64  	 */
65  	public boolean REST_SSL = false;
66  	/**
67  	 * SERVER REST interface using time limit (default: no limit <= 0)
68  	 */
69  	public long REST_TIME_LIMIT = -1;
70  	/**
71  	 * SERVER REST interface using authentication
72  	 */
73  	public boolean REST_AUTHENTICATED = false;
74  	/**
75  	 * SERVER REST interface using signature
76  	 */
77  	public boolean REST_SIGNATURE = true;
78  	/**
79  	 * Key for signature in SHA-256
80  	 *
81  	 */
82  	public HmacSha256 hmacSha256 = null;
83  	/**
84  	 * SERVER REST interface allowing one Handler and associated CRUD (or equivalent POST, GET, PUT, DELETE) methods 
85  	 * (2^0 for active, 2^1 as Create/POST, 2^2 as Read/GET, 2^3 as Update/PUT, 2^4 as Delete/DELETE)
86  	 */
87  	public byte [] RESTHANDLERS_CRUD = null;
88  
89  	/**
90  	 * Associated RestMethod Handlers
91  	 */
92  	public HashMap<String, RestMethodHandler> restHashMap = new HashMap<String, RestMethodHandler>();
93  	
94  	/**
95  	 * Set Key from String directly
96  	 * @param authentKey
97  	 */
98  	public void initializeKey(String authentKey) {
99  		hmacSha256 = new HmacSha256();
100 		hmacSha256.setSecretKey(authentKey.getBytes(WaarpStringUtils.UTF8));
101 	}
102 	/**
103 	 * Set Key from file
104 	 * @param authentKey
105 	 * @throws CryptoException
106 	 * @throws IOException
107 	 */
108 	public void initializeKey(File authentKey) throws CryptoException, IOException {
109 		hmacSha256 = new HmacSha256();
110 		hmacSha256.setSecretKey(authentKey);
111 	}
112 	
113 	public String toString() {
114 		String result = "{address: "+REST_ADDRESS+", port: "+REST_PORT+", ssl: "+REST_SSL+", time: "+REST_TIME_LIMIT+
115 				", authent:"+REST_AUTHENTICATED+", signature: "+REST_SIGNATURE+", handlers: [";
116 		for (Entry<String, RestMethodHandler> elt : restHashMap.entrySet()) {
117 			result += elt.getKey()+"="+elt.getValue().methods+", ";
118 		}
119 		result += "], crud: [";
120 		for (byte crud : RESTHANDLERS_CRUD) {
121 			result += (int) crud+", ";
122 		}
123 		result += "] }";
124 		return result;
125 	}
126 }

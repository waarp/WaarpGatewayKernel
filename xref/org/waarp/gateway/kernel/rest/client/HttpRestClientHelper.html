88  	/**
89  	 * @param baseUri base of all URI, in general simply "/" (default if null)
90  	 * @param nbclient max number of client connected at once
91  	 * @param timeout timeout used in connection
92  	 * @param pipelineFactory the associated client pipeline factory
93  	 */
94  	public HttpRestClientHelper(String baseUri, int nbclient, long timeout, ChannelPipelineFactory pipelineFactory) {
95  		if (logger == null) {
96  			logger = WaarpInternalLoggerFactory.getLogger(HttpRestClientHelper.class);
97  		}
98  		if (baseUri != null) {
99  			this.baseUri = baseUri;
100 		}
101 		channelClientFactory = new NioClientSocketChannelFactory(
102 				execServerBoss,
103 				execServerWorker,
104 				nbclient);
105 		clientBootstrap = new ClientBootstrap(channelClientFactory);
106 		clientBootstrap.setPipelineFactory(pipelineFactory);
107 		clientBootstrap.setOption("tcpNoDelay", true);
108 		clientBootstrap.setOption("reuseAddress", true);
109 		clientBootstrap.setOption("connectTimeoutMillis", timeout);
110 		// will ignore real request
111         HttpRequest request = new DefaultHttpRequest(HttpVersion.HTTP_1_1,
112                 HttpMethod.GET, baseUri);
113         headers = request.headers();
114         headers.set(HttpHeaders.Names.ACCEPT_ENCODING,
115                 HttpHeaders.Values.GZIP + "," + HttpHeaders.Values.DEFLATE);
116         headers.set(HttpHeaders.Names.ACCEPT_CHARSET,
117                 "utf-8;q=0.7,*;q=0.7");
118         headers.set(HttpHeaders.Names.ACCEPT_LANGUAGE, "fr,en");
119         headers.set(HttpHeaders.Names.USER_AGENT,
120                 "Netty Simple Http Rest Client side");
121         headers.set(HttpHeaders.Names.ACCEPT,
122                 "text/html,text/plain,application/xhtml+xml,application/xml,application/json;q=0.9,*/*;q=0.8");
123         // connection will not close but needed
124         /*request.headers().set(HttpHeaders.Names.CONNECTION,
125         HttpHeaders.Values.KEEP_ALIVE);*/
126         // request.setHeader("Connection","keep-alive");
127         // request.setHeader("Keep-Alive","300");
128 	}

132 	/**
133 	 * Create one new connection to the remote host using port
134 	 * @param host
135 	 * @param port
136 	 * @return the channel if connected or Null if not
137 	 */
138 	public Channel getChannel(String host, int port) {
139 		 // Start the connection attempt.
140         ChannelFuture future = clientBootstrap.connect(new InetSocketAddress(host, port));
141         // Wait until the connection attempt succeeds or fails.
142         Channel channel = WaarpSslUtility.waitforChannelReady(future);
143         if (channel != null) {
144         	RestFuture futureChannel = new RestFuture(true);
145         	channel.setAttachment(futureChannel);
146         }
147         return channel;
148 	}

150 	/**
151 	 * Send an HTTP query using the channel for target, using signature
152 	 * @param hmacSha256 SHA-256 key to create the signature
153 	 * @param channel target of the query
154 	 * @param method HttpMethod to use
155 	 * @param host target of the query (shall be the same as for the channel)
156 	 * @param addedUri additional uri, added to baseUri (shall include also extra arguments) (might be null)
157 	 * @param user user to use in authenticated Rest procedure (might be null)
158 	 * @param pwd password to use in authenticated Rest procedure (might be null)
159 	 * @param uriArgs arguments for Uri if any (might be null)
160 	 * @param json json to send as body in the request (might be null); Useful in PUT, POST but should not in GET, DELETE, OPTIONS
161 	 * @return the RestFuture associated with this request
162 	 */
163 	public RestFuture sendQuery(HmacSha256 hmacSha256, Channel channel, HttpMethod method, String host, String addedUri, String user, String pwd, Map<String, String> uriArgs, String json) {
164 		// Prepare the HTTP request.
165 		logger.debug("Prepare request: "+method+":"+addedUri+":"+json);
166 		RestFuture future = ((RestFuture) channel.getAttachment());
167         QueryStringEncoder encoder = null;
168         if (addedUri != null) {
169         	encoder = new QueryStringEncoder(baseUri+addedUri);
170         } else {
171         	encoder = new QueryStringEncoder(baseUri);
172         }
173         // add Form attribute
174         if (uriArgs != null) {
175         	for (Entry<String, String> elt : uriArgs.entrySet()) {
176 				encoder.addParam(elt.getKey(), elt.getValue());
177 			}
178         }
179         String [] result = null;
180         try {
181 			result = RestArgument.getBaseAuthent(hmacSha256, encoder, user, pwd);
182 			logger.debug("Authent encoded");
183 		} catch (HttpInvalidAuthenticationException e) {
184 			logger.error(e.getMessage(), e);
185 			future.setFailure(e);
186             return future;
187 		}
188         URI uri;
189 		try {
190 			uri = encoder.toUri();
191 		} catch (URISyntaxException e) {
192             logger.error(e.getMessage());
193             future.setFailure(e);
194             return future;
195         }
196 		logger.debug("Uri ready: "+uri.toASCIIString());
197 
198         HttpRequest request = new DefaultHttpRequest(HttpVersion.HTTP_1_1,
199                 method, uri.toASCIIString());
200         // it is legal to add directly header or cookie into the request until finalize
201         request.headers().add(this.headers);
202         request.headers().set(HttpHeaders.Names.HOST, host);
203         if (user != null) {
204             request.headers().set(RestArgument.REST_ROOT_FIELD.ARG_X_AUTH_USER.field, user);
205         }
206         request.headers().set(RestArgument.REST_ROOT_FIELD.ARG_X_AUTH_TIMESTAMP.field, result[0]);
207         request.headers().set(RestArgument.REST_ROOT_FIELD.ARG_X_AUTH_KEY.field, result[1]);
208         if (json != null) {
209     		logger.debug("Add body");
210         	ChannelBuffer buffer = ChannelBuffers.wrappedBuffer(json.getBytes(WaarpStringUtils.UTF8));
211             request.setContent(buffer);
212             request.headers().set(HttpHeaders.Names.CONTENT_LENGTH, buffer.readableBytes());
213         }
214         // send request
215 		logger.debug("Send request");
216 		channel.write(request);
217 		logger.debug("Request sent");
218 		return future;
219 	}

221 	/**
222 	 * Send an HTTP query using the channel for target, but without any Signature
223 	 * @param channel target of the query
224 	 * @param method HttpMethod to use
225 	 * @param host target of the query (shall be the same as for the channel)
226 	 * @param addedUri additional uri, added to baseUri (shall include also extra arguments) (might be null)
227 	 * @param user user to use in authenticated Rest procedure (might be null)
228 	 * @param uriArgs arguments for Uri if any (might be null)
229 	 * @param json json to send as body in the request (might be null); Useful in PUT, POST but should not in GET, DELETE, OPTIONS
230 	 * @return the RestFuture associated with this request
231 	 */
232 	public RestFuture sendQuery(Channel channel, HttpMethod method, String host, String addedUri, String user, Map<String, String> uriArgs, String json) {
233 		// Prepare the HTTP request.
234 		logger.debug("Prepare request: "+method+":"+addedUri+":"+json);
235 		RestFuture future = ((RestFuture) channel.getAttachment());
236         QueryStringEncoder encoder = null;
237         if (addedUri != null) {
238         	encoder = new QueryStringEncoder(baseUri+addedUri);
239         } else {
240         	encoder = new QueryStringEncoder(baseUri);
241         }
242         // add Form attribute
243         if (uriArgs != null) {
244         	for (Entry<String, String> elt : uriArgs.entrySet()) {
245 				encoder.addParam(elt.getKey(), elt.getValue());
246 			}
247         }
248         URI uri;
249 		try {
250 			uri = encoder.toUri();
251 		} catch (URISyntaxException e) {
252             logger.error(e.getMessage());
253             future.setFailure(e);
254             return future;
255         }
256 		logger.debug("Uri ready: "+uri.toASCIIString());
257 
258         HttpRequest request = new DefaultHttpRequest(HttpVersion.HTTP_1_1,
259                 method, uri.toASCIIString());
260         // it is legal to add directly header or cookie into the request until finalize
261         request.headers().add(this.headers);

263         if (user != null) {
264             request.headers().set(RestArgument.REST_ROOT_FIELD.ARG_X_AUTH_USER.field, user);
265         }
266         request.headers().set(RestArgument.REST_ROOT_FIELD.ARG_X_AUTH_TIMESTAMP.field, new DateTime().toString());
267         if (json != null) {
268     		logger.debug("Add body");
269         	ChannelBuffer buffer = ChannelBuffers.wrappedBuffer(json.getBytes(WaarpStringUtils.UTF8));
270             request.setContent(buffer);
271             request.headers().set(HttpHeaders.Names.CONTENT_LENGTH, buffer.readableBytes());
272         }
273         // send request
274 		logger.debug("Send request");
275 		channel.write(request);
276 		logger.debug("Request sent");
277 		return future;
278 	}

288 	/**
289 	 * 
290 	 * @param args as uri (http://host:port/uri method user pwd sign=path|nosign [json])
291 	 */
292 	public static void main(String[] args) {
293 		InternalLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
294         final WaarpInternalLogger logger = WaarpInternalLoggerFactory.getLogger(HttpRestClientHelper.class);
295 		if (args.length < 5) {
296 			logger.error("Need more arguments: http://host:port/uri method user pwd sign=path|nosign [json]");
297 			return;
298 		}
299 		String uri = args[0];
300 		String meth = args[1];
301 		String user = args[2];
302 		String pwd = args[3];
303 		boolean sign = args[4].toLowerCase().contains("sign=");
304 		HmacSha256 hmacSha256 = null;
305 		if (sign) {
306 			String file = args[4].replace("sign=", "");
307 			hmacSha256 = new HmacSha256();
308 			try {
309 				hmacSha256.setSecretKey(new File(file));
310 			} catch (CryptoException e) {
311 				logger.error("Need more arguments: http://host:port/uri method user pwd sign=path|nosign [json]");
312 				return;
313 			} catch (IOException e) {
314 				logger.error("Need more arguments: http://host:port/uri method user pwd sign=path|nosign [json]");
315 				return;
316 			}
317 		}
318 		String json = null;
319 		if (args.length > 5) {
320 			json = args[5].replace("'", "\"");
321 		}
322 		HttpMethod method = HttpMethod.valueOf(meth);
323 		int port = -1;
324 		String host = null;
325 		String path = null;
326 		try {
327 			URI realUri = new URI(uri);
328 			port = realUri.getPort();
329 			host = realUri.getHost();
330 			path = realUri.getPath();
331 		} catch (URISyntaxException e) {
332 			logger.error("Error", e);
333 			return;
334 		}
335 		HttpRestClientHelper client = new HttpRestClientHelper(path, 1, 30000, new HttpRestClientSimplePipelineFactory());
336 		Channel channel = client.getChannel(host, port);
337 		if (channel == null) {
338 			client.closeAll();
339 			logger.error("Cannot connect to "+host+" on port "+port);
340 			return;
341 		}
342 		RestFuture future = null;
343 		if (sign) {
344 			future = client.sendQuery(hmacSha256, channel, method, host, null, user, pwd, null, json);
345 		} else {
346 			future = client.sendQuery(channel, method, host, null, user, null, json);
347 		}
348 		try {
349 			future.await();
350 		} catch (InterruptedException e) {
351 			client.closeAll();
352 			logger.error("Interruption", e);
353 			return;
354 		}
355    		WaarpSslUtility.closingSslChannel(channel);
356 		if (future.isSuccess()) {
357 			logger.warn(future.getRestArgument().prettyPrint());
358 		} else {
359 			RestArgument ra = future.getRestArgument();
360 			if (ra != null) {
361 				logger.error(ra.prettyPrint());
362 			} else {
363 				logger.error("Query in error", future.getCause());
364 			}
365 		}
366 		client.closeAll();
367 	}

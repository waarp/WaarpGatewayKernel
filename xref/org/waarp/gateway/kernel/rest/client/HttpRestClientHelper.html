84      /**
85       * @param baseUri
86       *            base of all URI, in general simply "/" (default if null)
87       * @param nbclient
88       *            max number of client connected at once
89       * @param timeout
90       *            timeout used in connection
91       * @param Initializer
92       *            the associated client pipeline factory
93       */
94      public HttpRestClientHelper(String baseUri, int nbclient, long timeout,
95              ChannelInitializer<SocketChannel> Initializer) {
96          if (logger == null) {
97              logger = WaarpLoggerFactory.getLogger(HttpRestClientHelper.class);
98          }
99          if (baseUri != null) {
100             this.baseUri = baseUri;
101         }
102         // Configure the client.
103         bootstrap = new Bootstrap();
104         workerGroup = new NioEventLoopGroup(nbclient, new WaarpThreadFactory("Rest_" + baseUri + "_"));
105         WaarpNettyUtil.setBootstrap(bootstrap, workerGroup, 30000);
106         // Configure the pipeline factory.
107         bootstrap.handler(Initializer);
108 
109         // will ignore real request
110         HttpRequest request = new DefaultHttpRequest(HttpVersion.HTTP_1_1,
111                 HttpMethod.GET, baseUri);
112         headers = request.headers();
113         headers.set(HttpHeaderNames.ACCEPT_ENCODING,
114                 HttpHeaderValues.GZIP + "," + HttpHeaderValues.DEFLATE);
115         headers.set(HttpHeaderNames.ACCEPT_CHARSET,
116                 "utf-8;q=0.7,*;q=0.7");
117         headers.set(HttpHeaderNames.ACCEPT_LANGUAGE, "fr,en");
118         headers.set(HttpHeaderNames.USER_AGENT,
119                 "Netty Simple Http Rest Client side");
120         headers.set(HttpHeaderNames.ACCEPT,
121                 "text/html,text/plain,application/xhtml+xml,application/xml,application/json;q=0.9,*/*;q=0.8");
122         // connection will not close but needed
123         /*request.headers().set(HttpHeaders.Names.CONNECTION,
124         HttpHeaders.Values.KEEP_ALIVE);*/
125         // request.setHeader("Connection","keep-alive");
126         // request.setHeader("Keep-Alive","300");
127     }

129     /**
130      * Create one new connection to the remote host using port
131      * 
132      * @param host
133      * @param port
134      * @return the channel if connected or Null if not
135      */
136     public Channel getChannel(String host, int port) {
137         // Start the connection attempt.
138         ChannelFuture future = bootstrap.connect(new InetSocketAddress(host, port));
139         // Wait until the connection attempt succeeds or fails.
140         Channel channel = WaarpSslUtility.waitforChannelReady(future);
141         if (channel != null) {
142             RestFuture futureChannel = new RestFuture(true);
143             channel.attr(HttpRestClientSimpleResponseHandler.RESTARGUMENT).set(futureChannel);
144         }
145         return channel;
146     }

148     /**
149      * Send an HTTP query using the channel for target, using signature
150      * 
151      * @param hmacSha256
152      *            SHA-256 key to create the signature
153      * @param channel
154      *            target of the query
155      * @param method
156      *            HttpMethod to use
157      * @param host
158      *            target of the query (shall be the same as for the channel)
159      * @param addedUri
160      *            additional uri, added to baseUri (shall include also extra arguments) (might be null)
161      * @param user
162      *            user to use in authenticated Rest procedure (might be null)
163      * @param pwd
164      *            password to use in authenticated Rest procedure (might be null)
165      * @param uriArgs
166      *            arguments for Uri if any (might be null)
167      * @param json
168      *            json to send as body in the request (might be null); Useful in PUT, POST but should not in GET, DELETE, OPTIONS
169      * @return the RestFuture associated with this request
170      */
171     public RestFuture sendQuery(HmacSha256 hmacSha256, Channel channel, HttpMethod method, String host,
172             String addedUri, String user, String pwd, Map<String, String> uriArgs, String json) {
173         // Prepare the HTTP request.
174         logger.debug("Prepare request: " + method + ":" + addedUri + ":" + json);
175         RestFuture future = channel.attr(HttpRestClientSimpleResponseHandler.RESTARGUMENT).get();
176         QueryStringEncoder encoder = null;
177         if (addedUri != null) {
178             encoder = new QueryStringEncoder(baseUri + addedUri);
179         } else {
180             encoder = new QueryStringEncoder(baseUri);
181         }
182         // add Form attribute
183         if (uriArgs != null) {
184             for (Entry<String, String> elt : uriArgs.entrySet()) {
185                 encoder.addParam(elt.getKey(), elt.getValue());
186             }
187         }
188         String[] result = null;
189         try {
190             result = RestArgument.getBaseAuthent(hmacSha256, encoder, user, pwd);
191             logger.debug("Authent encoded");
192         } catch (HttpInvalidAuthenticationException e) {
193             logger.error(e.getMessage(), e);
194             future.setFailure(e);
195             return future;
196         }
197         URI uri;
198         try {
199             uri = encoder.toUri();
200         } catch (URISyntaxException e) {
201             logger.error(e.getMessage());
202             future.setFailure(e);
203             return future;
204         }
205         logger.debug("Uri ready: " + uri.toASCIIString());
206 
207         FullHttpRequest request;
208         if (json != null) {
209             logger.debug("Add body");
210             ByteBuf buffer = Unpooled.wrappedBuffer(json.getBytes(WaarpStringUtils.UTF8));
211             request = new DefaultFullHttpRequest(HttpVersion.HTTP_1_1, method, uri.toASCIIString(), buffer);
212             request.headers().set(HttpHeaderNames.CONTENT_LENGTH, buffer.readableBytes());
213         } else {
214             request = new DefaultFullHttpRequest(HttpVersion.HTTP_1_1, method, uri.toASCIIString());
215         }
216         // it is legal to add directly header or cookie into the request until finalize
217         request.headers().add(this.headers);
218         request.headers().set(HttpHeaderNames.HOST, host);
219         if (user != null) {
220             request.headers().set((CharSequence) RestArgument.REST_ROOT_FIELD.ARG_X_AUTH_USER.field, user);
221         }
222         request.headers().set((CharSequence) RestArgument.REST_ROOT_FIELD.ARG_X_AUTH_TIMESTAMP.field, result[0]);
223         request.headers().set((CharSequence) RestArgument.REST_ROOT_FIELD.ARG_X_AUTH_KEY.field, result[1]);
224         // send request
225         logger.debug("Send request");
226         channel.writeAndFlush(request);
227         logger.debug("Request sent");
228         return future;
229     }

231     /**
232      * Send an HTTP query using the channel for target, but without any Signature
233      * 
234      * @param channel
235      *            target of the query
236      * @param method
237      *            HttpMethod to use
238      * @param host
239      *            target of the query (shall be the same as for the channel)
240      * @param addedUri
241      *            additional uri, added to baseUri (shall include also extra arguments) (might be null)
242      * @param user
243      *            user to use in authenticated Rest procedure (might be null)
244      * @param uriArgs
245      *            arguments for Uri if any (might be null)
246      * @param json
247      *            json to send as body in the request (might be null); Useful in PUT, POST but should not in GET, DELETE, OPTIONS
248      * @return the RestFuture associated with this request
249      */
250     public RestFuture sendQuery(Channel channel, HttpMethod method, String host, String addedUri, String user,
251             Map<String, String> uriArgs, String json) {
252         // Prepare the HTTP request.
253         logger.debug("Prepare request: " + method + ":" + addedUri + ":" + json);
254         RestFuture future = channel.attr(HttpRestClientSimpleResponseHandler.RESTARGUMENT).get();
255         QueryStringEncoder encoder = null;
256         if (addedUri != null) {
257             encoder = new QueryStringEncoder(baseUri + addedUri);
258         } else {
259             encoder = new QueryStringEncoder(baseUri);
260         }
261         // add Form attribute
262         if (uriArgs != null) {
263             for (Entry<String, String> elt : uriArgs.entrySet()) {
264                 encoder.addParam(elt.getKey(), elt.getValue());
265             }
266         }
267         URI uri;
268         try {
269             uri = encoder.toUri();
270         } catch (URISyntaxException e) {
271             logger.error(e.getMessage());
272             future.setFailure(e);
273             return future;
274         }
275         logger.debug("Uri ready: " + uri.toASCIIString());
276 
277         FullHttpRequest request;
278         if (json != null) {
279             logger.debug("Add body");
280             ByteBuf buffer = Unpooled.wrappedBuffer(json.getBytes(WaarpStringUtils.UTF8));
281             request = new DefaultFullHttpRequest(HttpVersion.HTTP_1_1, method, uri.toASCIIString(), buffer);
282             request.headers().set(HttpHeaderNames.CONTENT_LENGTH, buffer.readableBytes());
283         } else {
284             request = new DefaultFullHttpRequest(HttpVersion.HTTP_1_1, method, uri.toASCIIString());
285         }
286 
287         // it is legal to add directly header or cookie into the request until finalize
288         request.headers().add(this.headers);
289         request.headers().set(HttpHeaderNames.HOST, host);
290         if (user != null) {
291             request.headers().set((CharSequence) RestArgument.REST_ROOT_FIELD.ARG_X_AUTH_USER.field, user);
292         }
293         request.headers().set((CharSequence) RestArgument.REST_ROOT_FIELD.ARG_X_AUTH_TIMESTAMP.field, new DateTime().toString());
294         // send request
295         logger.debug("Send request");
296         channel.writeAndFlush(request);
297         logger.debug("Request sent");
298         return future;
299     }

308     /**
309      * 
310      * @param args
311      *            as uri (http://host:port/uri method user pwd sign=path|nosign [json])
312      */
313     public static void main(String[] args) {
314         WaarpLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
315         final WaarpLogger logger = WaarpLoggerFactory.getLogger(HttpRestClientHelper.class);
316         if (args.length < 5) {
317             logger.error("Need more arguments: http://host:port/uri method user pwd sign=path|nosign [json]");
318             return;
319         }
320         String uri = args[0];
321         String meth = args[1];
322         String user = args[2];
323         String pwd = args[3];
324         boolean sign = args[4].toLowerCase().contains("sign=");
325         HmacSha256 hmacSha256 = null;
326         if (sign) {
327             String file = args[4].replace("sign=", "");
328             hmacSha256 = new HmacSha256();
329             try {
330                 hmacSha256.setSecretKey(new File(file));
331             } catch (CryptoException e) {
332                 logger.error("Need more arguments: http://host:port/uri method user pwd sign=path|nosign [json]");
333                 return;
334             } catch (IOException e) {
335                 logger.error("Need more arguments: http://host:port/uri method user pwd sign=path|nosign [json]");
336                 return;
337             }
338         }
339         String json = null;
340         if (args.length > 5) {
341             json = args[5].replace("'", "\"");
342         }
343         HttpMethod method = HttpMethod.valueOf(meth);
344         int port = -1;
345         String host = null;
346         String path = null;
347         try {
348             URI realUri = new URI(uri);
349             port = realUri.getPort();
350             host = realUri.getHost();
351             path = realUri.getPath();
352         } catch (URISyntaxException e) {
353             logger.error("Error", e);
354             return;
355         }
356         HttpRestClientHelper client = new HttpRestClientHelper(path, 1, 30000, new HttpRestClientSimpleInitializer());
357         Channel channel = client.getChannel(host, port);
358         if (channel == null) {
359             client.closeAll();
360             logger.error("Cannot connect to " + host + " on port " + port);
361             return;
362         }
363         RestFuture future = null;
364         if (sign) {
365             future = client.sendQuery(hmacSha256, channel, method, host, null, user, pwd, null, json);
366         } else {
367             future = client.sendQuery(channel, method, host, null, user, null, json);
368         }
369         try {
370             future.await();
371         } catch (InterruptedException e) {
372             client.closeAll();
373             logger.error("Interruption", e);
374             return;
375         }
376         WaarpSslUtility.closingSslChannel(channel);
377         if (future.isSuccess()) {
378             logger.warn(future.getRestArgument().prettyPrint());
379         } else {
380             RestArgument ra = future.getRestArgument();
381             if (ra != null) {
382                 logger.error(ra.prettyPrint());
383             } else {
384                 logger.error("Query in error", future.getCause());
385             }
386         }
387         client.closeAll();
388     }
